DCL-PI ORDER_RETRIEVE;
  OrderID ZONED(10:0);
  OrderDS LIKEDS(OrderHeader);

// Data Structure
DCL-DS OrderHeader QUALIFIED;
  OrderID ZONED(10:0);
  CustID ZONED(10:0);
  TotalAmt PACKED(15:2);
  Status CHAR(10);
END-DS;

// Files
DCL-F ORDERS USROPN KEYED;

// Variables
DCL-S I INT(10);
DCL-S Temp PACKED(15:2);

// Open
OPEN ORDERS;

// Logic
MONITOR;
  CHAIN OrderID ORDERS;
  IF %FOUND;
    OrderDS.OrderID = ORDERID;
    OrderDS.TotalAmt = TOTALAMT;
    OrderDS.Status = STATUS;
  ELSE;
    DSPLY 'Order not found';
    RETURN;
  ENDIF;
ON-ERROR;
  DSPLY 'Retrieve error';
ENDMON;
