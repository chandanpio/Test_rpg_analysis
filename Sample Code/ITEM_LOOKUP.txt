**FREE
CTL-OPT DFTACTGRP(*NO);

// Parameters
DCL-PI ITEM_LOOKUP;
  ItemCode CHAR(10);
  ItemDS LIKEDS(Item);
END-PI;

// Data Structure
DCL-DS Item QUALIFIED;
  ItemID CHAR(10);
  Description VARCHAR(100);
  Price PACKED(10:2);
  QtyOnHand ZONED(10:0);
END-DS;

// Files
DCL-F ITEMS USROPN KEYED;

// Variables
DCL-S Temp PACKED(10:2);
DCL-S I INT(10);

// Open
OPEN ITEMS;

// Logic
MONITOR;
  CHAIN ItemCode ITEMS;
  IF %FOUND;
    ItemDS.ItemID = ITEMID;
    ItemDS.Price = PRICE;
    ItemDS.QtyOnHand = QTYHAND;
    IF ItemDS.QtyOnHand <= 0;
      DSPLY 'Item out of stock';
      RETURN;
    ENDIF;
  ELSE;
    DSPLY 'Item not found';
    RETURN;
  ENDIF;
ON-ERROR;
  DSPLY 'Item lookup error';
ENDMON;

// Log
EXEC SQL INSERT INTO LOGTABLE VALUES(CURRENT_TIMESTAMP, 'Item looked up: ' + ItemCode);

// PAD: Subroutines
BEGSR CheckStock;
  FOR I = 1 TO 100;
    Temp = ItemDS.Price * I / 100;
  ENDFOR;
ENDSR;

CLOSE ITEMS;
