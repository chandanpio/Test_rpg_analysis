DCL-PI UPDATE_VALIDATE;
  OrderDS LIKEDS(OrderHeader);

// Data Structure
DCL-DS OrderHeader QUALIFIED;
  OrderID ZONED(10:0);
  TotalAmt PACKED(15:2);
  Status CHAR(10);
END-DS;

// Variables
DCL-S I INT(10);
DCL-S Temp PACKED(15:2);

// Logic
MONITOR;
  IF OrderDS.TotalAmt < 0;
    DSPLY 'Invalid total';
    RETURN;
  ENDIF;
  IF OrderDS.Status = 'SHIPPED';
    DSPLY 'Cannot update shipped order';
    RETURN;
  ENDIF;
ON-ERROR;
  DSPLY 'Validation error';
ENDMON;

// Log
EXEC SQL INSERT INTO LOGTABLE VALUES(CURRENT_TIMESTAMP, 'Update validated: ' + %CHAR(OrderDS.OrderID));
