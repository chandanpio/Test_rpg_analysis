DCL-PI ORDER_CALC;

// Data Structure
DCL-DS OrderHeader QUALIFIED;
  OrderID ZONED(10:0);
  CustID ZONED(10:0);
  TotalAmt PACKED(15:2);
  Discount PACKED(5:2);
  TaxAmt PACKED(15:2);
END-DS;

// Variables
DCL-S TaxRate PACKED(5:2) INZ(0.08);
DCL-S I INT(10);
DCL-S Temp PACKED(15:2);

// Logic
MONITOR;
  OrderDS.Discount = OrderDS.TotalAmt * 0.05;  // 5% discount
  OrderDS.TaxAmt = (OrderDS.TotalAmt - OrderDS.Discount) * TaxRate;
  OrderDS.TotalAmt += OrderDS.TaxAmt;
ON-ERROR;
  DSPLY 'Calculation error';
ENDMON;
